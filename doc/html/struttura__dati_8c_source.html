63 if(prima < minimo) {
64 // Imposta un nuovo minimo
65 minimo = prima;

67 // Se la colonna calcolata e' superiore al massimo
68 if(ultima > massimo) {
69 // Imposta un nuovo massimo
70 massimo = ultima;
71 }
72 }
73 // Stampa un intestazione contenente le colonne
74 printf(" ");
75 for(unsigned int j=minimo; j<=massimo; j++) {
76 printf(" %2u ", j);
77 }
78 printf("\n");
79 // Per ogni riga della piano_gioco
80 for(unsigned int i=0; i<piano_gioco->righe; i++) {
81 // Stampa l'indice della riga corrente

105void stampa_estremi(estremo_t *estremi, size_t massimo, size_t minimo) {
106 // Per ciascun estremo nella struttura dati
107 for(size_t i = minimo; i <= massimo; i++) {
108 // In base al valore assunto dal cardine
109 switch(estremi[i].cardine) {
110 // Stampa correttamente l'estremo
111 case 1: printf(" [%u ", estremi[i].valore); // >
112 break;
113 case 2: printf(" %u] ", estremi[i].valore); // <
114 break;
115 case 3: printf(" {%u ", estremi[i].valore); // v
116 break;
117 case 4: printf(" %u} ", estremi[i].valore); // ^
118 break;
119 // Quando non e' stato memorizzato un valore stampa uno spazio vuoto
120 default: printf(" -- ");
121 }
122 }
123 // Stampa il carattere per andare a capo
124 printf("\n");
125}
