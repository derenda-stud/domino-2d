29bool posizione_valida(matrice_t *piano_gioco, coord_t coordinata, bool orientamento) {
30 // Controlla che la posizione attuale sia libera
31 if (piano_gioco->posizione[coordinata.riga][coordinata.colonna].cardine) return false;
32 // Per le tessere posizionate in orizzontale
33 if (orientamento) {
34 // Controlla che la posizione successiva sia libera
35 if (piano_gioco->posizione[coordinata.riga][coordinata.colonna + 1].cardine) return false;
36 // Altrimenti per quelle posizionate in verticale
37 } else {
38 // Controlla di rientrare nella riga e che la posizione sottostante sia libera
39 if (piano_gioco->righe > coordinata.riga + 1 && piano_gioco->posizione[coordinata.riga + 1][coordinata.colonna].cardine) return false;
40 }
41 // Controlla che abbia almeno un collegamento tra l'estremo sinistro e quello destro
42 if (!piano_gioco->posizione[coordinata.riga][coordinata.colonna - 1].cardine && !piano_gioco->posizione[coordinata.riga][coordinata.colonna + orientamento + 1].cardine) return false;
43 // Per tutti i casi rimanenti la posizione risulta valida
44 return true;
45}

114bool mossa_legale(matrice_t *piano_gioco, coord_t *coordinata, tessera_t *tessera, comb_t *risultato) {
115 // Se l'unica coordinate presente e' quella centrale
116 if (coordinata->riga == 0 && coordinata->colonna == piano_gioco->colonne / 2 - 1) {
117 // Controlla che non sia stata inserita una speciale diversa da [0|0]
118 if (tessera->speciale && tessera->sinistro != 0) return false;
119 // Altrimenti l'inserimento e' sempre valido
120 return true;
121 }
122 // Controlla che la tessera da inserire abbia almeno una corrispondenza
123 int corrente = estremi_corrispondono(piano_gioco, *coordinata, *tessera, risultato);
124 // Imposta il valore della variabile booleana a [0/1]
125 risultato->rotazione = corrente - 1;
126 // Quando sono in orizzontale, non ho corrispondenza, oppure sono arrivato all'ultima riga
127 if (risultato->orientamento || !corrente || piano_gioco->righe == coordinata->riga + 1) {
128 // Salva solamente il controllo della la riga corrente
129 return corrente;
130 }
131 // Altrimenti controlla anche la corrispondenza della riga sottostante
132 int sottostante = estremi_corrispondono(piano_gioco, (coord_t){coordinata->riga + 1, coordinata->colonna}, (tessera_t){tessera->destro, tessera->sinistro, tessera->speciale}, risultato);
133 // Rimane la corrispondenza se viene rispettato l'ordine
134 if (corrente == sottostante) return sottostante;
135 // Per i casi verticali rimanenti non e' possibile posizionarla in verticale
136 return false;
137}

139int estremi_corrispondono(matrice_t *piano_gioco, coord_t coordinata, tessera_t tessera, comb_t *risultato) {
140 // Recupera l'indirizzo dell'estremo del piano alla posizione indicata
141 estremo_t *estremo = &piano_gioco->posizione[coordinata.riga][coordinata.colonna];
142 // Se inserisco una speciale oppure entrambi gli estremi appartengono alla [0|0] mantieni l'ordine
143 if (!(estremo - 1)->valore && !(estremo + risultato->orientamento + 1)->valore) return true;
144 // La tessera non potra' combaciare se gli estremi presenti sono discordanti
145 if ((estremo - 1)->cardine && (estremo + risultato->orientamento + 1)->cardine && (estremo - 1)->valore != (estremo + risultato->orientamento + 1)->valore) return false;
146 // Memorizza i valori di controllo
147 int controllo_sx = 0, controllo_dx = 0;
148 // Controlla che esista l'estremo precedente
149 if ((estremo - 1)->cardine) {
150 // Assegna la posizione precedente come coordinata adiacente
151 *(risultato->adiacente) = (coord_t){coordinata.riga, coordinata.colonna - 1};
152 // In caso di inserimento speciale mantieni l'ordine
153 if (tessera.speciale) controllo_sx = 1;
154 // Controlla l'inserimento mantenendo l'ordine
155 if ((estremo - 1)->valore == tessera.sinistro) controllo_sx = 1;
156 // Controlla l'inserimento girando la tessera
157 else if ((estremo - 1)->valore == tessera.destro) controllo_sx = 2;
158 }
159 // Controlla che esista l'estremo successivo
160 if ((estremo + risultato->orientamento + 1)->cardine) {
161 // Assegna la posizione successiva come coordinata adiacente
162 *(risultato->adiacente) = (coord_t){coordinata.riga, coordinata.colonna + risultato->orientamento + 1};
163 // In caso di inserimento speciale mantieni l'ordine
164 if (tessera.speciale) controllo_dx = 1;
165 /* In base all'orientamento della tessera l'estremo successivo corrisponde:
166 - al valore sinistro in orizzontale -> inserisci girando la tessera
167 - al valore sinistro in verticale -> inserisci mantenendo l'ordine
168 */
169 if ((estremo + risultato->orientamento + 1)->valore == tessera.sinistro) controllo_dx = 1 + risultato->orientamento;
170 /*
171 - al valore destro in orizzontale -> inserisci mantenendo l'ordine
172 - al valore destro in verticale -> inserisci girando la tessera
173 */
174 else if ((estremo + risultato->orientamento + 1)->valore == tessera.destro) controllo_dx = 1 + !risultato->orientamento;
175 }
176 // Il risultato del controllo e' stabilito tramite un'operazione bitwise
177 return (controllo_sx | controllo_dx) % 3;
178}
