6void genera_tessere(vect_t *mano_giocatore) {
7 // Calcola la percentuale di tessere speciali
8 int dimensione_speciali = mano_giocatore->capacita / 4;
9 // Per le posizioni in cui inserire tessere normali
10 for(size_t i=0; i<mano_giocatore->capacita - dimensione_speciali; i++) {
11 // Crea una tessera temporanea da copiare dentro la mano
12 tessera_t tessera_attuale;
13 // Genera casualmente i valori contenuti nella tessera
14 tessera_attuale.sinistro = rand() % 6 + 1;
15 tessera_attuale.destro = rand() % 6 + 1;
16 tessera_attuale.speciale = false;
17 // Inserisci la tessera dentro la mano del giocatore
18 inserimento_coda(mano_giocatore, &tessera_attuale);
19 }
20 // Struttura contenente le diverse tessere speciali
21 tessera_t tessere_speciali[3] = {{0, 0, true}, {11, 11, true}, {12, 21, true}};
22 // Per le posizioni rimanenti nella mano del giocatore
23 for(size_t i=0; i<dimensione_speciali; i++) {
24 // Seleziona casualmente una delle tessere speciali
25 tessera_t speciale = tessere_speciali[rand() % 3];
26 // Inserisci la tessera speciale generata nella mano
27 inserimento_coda(mano_giocatore, &speciale);
28 }
29}

31vect_t *calcola_coordinate(matrice_t *piano_gioco, bool orientamento) {
32 // Crea il vettore contenente le coordinate
33 vect_t *coordinate = crea_vettore(sizeof(coord_t), 0);
34 // Se non sono ancora state inserite tessere il piano di gioco e' vuoto
35 if(prima_posizione(piano_gioco, 0) > ultima_posizione(piano_gioco, 0)) {
36 // Calcola la coordinata della posizione centrale
37 coord_t centrale = {0, piano_gioco->colonne / 2 - 1};
38 // Inserisci come unica coordinata quella centrale
39 inserimento_coda(coordinate, &centrale);
40 // Non fare ulteriore inserimenti
41 return coordinate;
42 }
43 // Per ciascuna riga del piano di gioco
44 for(size_t i=0; i<piano_gioco->righe; i++) {
45 // Calcolo la prima e l'ultima posizione utile
46 unsigned int inizio = prima_posizione(piano_gioco, i) - orientamento;
47 unsigned int fine = ultima_posizione(piano_gioco, i);
48 // Scorri tutte le posizioni utili della riga corrente
49 for(size_t j=inizio; j<=fine; j++) {
50 // Memorizza la coordinata attuale
51 coord_t attuale = {i, j};
52 // Trova dove e' possibile posizionare una tessera
53 if(posizione_valida(piano_gioco, attuale, orientamento)) {
54 // Inserisci la nuova coordinata valida
55 inserimento_coda(coordinate, &attuale);
56 }
57 }
58 }
59 // Ritorna il nuovo vettore creato
60 return coordinate;
61}
